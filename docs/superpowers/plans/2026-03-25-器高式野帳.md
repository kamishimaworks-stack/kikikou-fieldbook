> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a mobile-first PWA for leveling survey field book data entry with reactive spreadsheet-like calculations.

**Architecture:** Single-page Vanilla JS app with two views (file list and field book editor) swapped via DOM manipulation. A pure-function calculation engine processes rows top-to-bottom on every change. LocalStorage persistence with auto-save on each edit.

**Tech Stack:** Vanilla HTML/CSS/JS, PWA (Service Worker + manifest.json), LocalStorage

```javascript
// calc.js
"use strict";

/**
 * Recalculate all rows top-to-bottom.
 * Each row: { id, point, bs, fs, gh, fh, ih, diff, manualFields }
 * - manualFields: array of field names manually entered (e.g. ["gh", "bs"])
 * - ih and diff are always computed (never in manualFields)
 * Returns a new array of rows with ih, diff, and non-manual gh filled in.
 */
function recalculate(rows) {
  let currentIH = null;

  return rows.map((row) => {
    const newRow = Object.assign({}, row);
    const manual = newRow.manualFields || [];

    // Step 1: Determine GH
    if (manual.includes("gh")) {
      // Manual GH — keep as-is
    } else if (newRow.fs !== null && currentIH !== null) {
      newRow.gh = roundM(currentIH - newRow.fs);
    } else {
      newRow.gh = null;
    }

    // Step 2: Determine IH
    if (newRow.bs !== null && newRow.gh !== null) {
      currentIH = roundM(newRow.gh + newRow.bs);
      newRow.ih = currentIH;
    } else {
      newRow.ih = null;
      // currentIH carries forward from previous row (no change)
    }

    // Step 3: Determine Diff
    if (newRow.gh !== null && newRow.fh !== null) {
      newRow.diff = roundM(newRow.gh - newRow.fh);
    } else {
      newRow.diff = null;
    }

    return newRow;
  });
}

/** Round to 3 decimal places (mm precision) to avoid floating point drift */
function roundM(value) {
  return Math.round(value * 1000) / 1000;
}

/** Format a value for display. Returns "" for null. */
function formatValue(value, displayMode) {
  if (value === null) return "";
  if (displayMode === "mm") {
    return String(Math.round(value * 1000));
  }
  return value.toFixed(3);
}

/** Parse input value to internal meters. Returns null for empty/invalid. */
function parseInput(str, displayMode) {
  if (str === "" || str === null || str === undefined) return null;
  const num = parseFloat(str);
  if (isNaN(num)) return null;
  if (displayMode === "mm") {
    return roundM(num / 1000);
  }
  return roundM(num);
}

/** Create a new empty row */
function createRow() {
  return {
    id: crypto.randomUUID(),
    point: "",
    bs: null,
    fs: null,
    gh: null,
    fh: null,
    ih: null,
    diff: null,
    manualFields: [],
  };
}
```

```javascript
// storage.js
"use strict";

const STORAGE_KEY = "kikikou_fieldbook_data";

/** Load all files from LocalStorage. Returns { files: { id: fileData, ... } } */
function loadAllFiles() {
  try {
    const raw = localStorage.getItem(STORAGE_KEY);
    if (!raw) return { files: {} };
    return JSON.parse(raw);
  } catch (e) {
    console.error("Failed to load data:", e);
    return { files: {} };
  }
}

/** Save entire data object to LocalStorage */
function saveAllFiles(data) {
  try {
    localStorage.setItem(STORAGE_KEY, JSON.stringify(data));
  } catch (e) {
    console.error("Failed to save data:", e);
    alert("保存に失敗しました。ストレージ容量が不足している可能性があります。");
  }
}

/** Get sorted file list (newest first) */
function getFileList() {
  const data = loadAllFiles();
  return Object.values(data.files).sort(
    (a, b) => new Date(b.updatedAt) - new Date(a.updatedAt)
  );
}

/** Create a new file and return it */
function createFile(name, displayMode) {
  const data = loadAllFiles();
  const id = crypto.randomUUID();
  const now = new Date().toISOString();
  const file = {
    id: id,
    name: name,
    displayMode: displayMode, // "m" or "mm"
    createdAt: now,
    updatedAt: now,
    rows: [createRow()], // Start with one empty row
  };
  data.files[id] = file;
  saveAllFiles(data);
  return file;
}

/** Load a single file by ID. Returns null if not found. */
function loadFile(fileId) {
  const data = loadAllFiles();
  return data.files[fileId] || null;
}

/** Save a single file (update in storage) */
function saveFile(file) {
  const data = loadAllFiles();
  file.updatedAt = new Date().toISOString();
  data.files[file.id] = file;
  saveAllFiles(data);
}

/** Delete a file by ID */
function deleteFile(fileId) {
  const data = loadAllFiles();
  delete data.files[fileId];
  saveAllFiles(data);
}

/** Rename a file */
function renameFile(fileId, newName) {
  const data = loadAllFiles();
  if (data.files[fileId]) {
    data.files[fileId].name = newName;
    data.files[fileId].updatedAt = new Date().toISOString();
    saveAllFiles(data);
  }
}
```

```javascript
// keypad.js
"use strict";

const Keypad = (function () {
  let _value = "";
  let _isNegative = false;
  let _onConfirm = null;
  let _displayEl = null;
  let _overlayEl = null;

  function init() {
    _overlayEl = document.getElementById("keypad-overlay");
    _displayEl = document.getElementById("keypad-display");

    // Number and dot keys
    document.querySelectorAll(".keypad-key").forEach((btn) => {
      btn.addEventListener("click", function () {
        const key = this.dataset.key;
        if (key === "backspace") {
          _value = _value.slice(0, -1);
        } else if (key === ".") {
          if (!_value.includes(".")) _value += ".";
        } else {
          _value += key;
        }
        _updateDisplay();
      });
    });

    // Sign toggle
    document.getElementById("keypad-sign").addEventListener("click", function () {
      _isNegative = !_isNegative;
      _updateDisplay();
    });

    // Done
    document.getElementById("keypad-done").addEventListener("click", function () {
      const result = _getFullValue();
      hide();
      if (_onConfirm) _onConfirm(result);
    });

    // Tap overlay background to cancel
    _overlayEl.addEventListener("click", function (e) {
      if (e.target === _overlayEl) {
        hide();
      }
    });
  }

  function show(currentValue, onConfirm) {
    _onConfirm = onConfirm;
    if (currentValue !== "" && currentValue !== null && currentValue !== undefined) {
      const num = parseFloat(currentValue);
      if (!isNaN(num)) {
        _isNegative = num < 0;
        _value = String(Math.abs(num));
      } else {
        _value = "";
        _isNegative = false;
      }
    } else {
      _value = "";
      _isNegative = false;
    }
    _updateDisplay();
    _overlayEl.classList.remove("hidden");
  }

  function hide() {
    _overlayEl.classList.add("hidden");
    _onConfirm = null;
  }

  function _getFullValue() {
    if (_value === "" || _value === ".") return "";
    return (_isNegative ? "-" : "") + _value;
  }

  function _updateDisplay() {
    const full = _getFullValue();
    _displayEl.textContent = full === "" ? "0" : full;
  }

  return { init: init, show: show, hide: hide };
})();
```

```javascript
// app.js
"use strict";

const App = (function () {
  let _currentFile = null;
  let _deletedRow = null; // For undo
  let _deletedRowIndex = -1;
  let _toastTimer = null;

  // ===== Init =====
  function init() {
    Keypad.init();
    _bindFileListEvents();
    _bindEditorEvents();
    _bindAutoSave();
    showFileList();
  }

  // ===== View Switching =====
  function showFileList() {
    _currentFile = null;
    document.getElementById("view-file-list").classList.remove("hidden");
    document.getElementById("view-editor").classList.add("hidden");
    _renderFileList();
  }

  function showEditor(fileId) {
    _currentFile = loadFile(fileId);
    if (!_currentFile) return;
    document.getElementById("view-file-list").classList.add("hidden");
    document.getElementById("view-editor").classList.remove("hidden");
    document.getElementById("editor-title").textContent = _currentFile.name;
    _updateModeButtons();
    _recalcAndRender();
  }

  // ===== File List =====
  function _renderFileList() {
    const container = document.getElementById("file-list-container");
    const files = getFileList();

    if (files.length === 0) {
      container.innerHTML = '<div class="empty-state"><p>ファイルがありません</p><p>「＋ 新規作成」で始めましょう</p></div>';
      return;
    }

    container.innerHTML = files
      .map((f) => {
        const date = new Date(f.updatedAt).toLocaleDateString("ja-JP");
        const mode = f.displayMode === "mm" ? "プラマイ(mm)" : "地盤高(m)";
        return '<div class="file-card" data-id="' + f.id + '">' +
          '<div class="file-info">' +
            '<h3>' + _escapeHtml(f.name) + '</h3>' +
            '<p>' + date + ' ・ ' + mode + ' ・ ' + f.rows.length + '行</p>' +
          '</div>' +
          '<div class="file-actions">' +
            '<button class="btn-rename" data-id="' + f.id + '">名前変更</button>' +
            '<button class="btn-delete" data-id="' + f.id + '">削除</button>' +
          '</div>' +
        '</div>';
      })
      .join("");
  }

  function _bindFileListEvents() {
    // Open file
    document.getElementById("file-list-container").addEventListener("click", function (e) {
      const card = e.target.closest(".file-card");
      if (!card) return;
      // Ignore if clicking action buttons
      if (e.target.closest(".file-actions")) {
        const btn = e.target.closest("button");
        if (!btn) return;
        const id = btn.dataset.id;
        if (btn.classList.contains("btn-rename")) {
          const newName = prompt("新しい現場名を入力してください");
          if (newName && newName.trim()) {
            renameFile(id, newName.trim());
            _renderFileList();
          }
        } else if (btn.classList.contains("btn-delete")) {
          if (confirm("「" + loadFile(id).name + "」を削除しますか？")) {
            deleteFile(id);
            _renderFileList();
          }
        }
        return;
      }
      showEditor(card.dataset.id);
    });

    // New file dialog
    document.getElementById("btn-new-file").addEventListener("click", function () {
      document.getElementById("input-file-name").value = "";
      document.getElementById("dialog-new-file").classList.remove("hidden");
    });
    document.getElementById("btn-cancel-new").addEventListener("click", function () {
      document.getElementById("dialog-new-file").classList.add("hidden");
    });
    document.getElementById("btn-confirm-new").addEventListener("click", function () {
      const name = document.getElementById("input-file-name").value.trim();
      if (!name) { alert("現場名を入力してください"); return; }
      const mode = document.querySelector('input[name="display-mode"]:checked').value;
      const file = createFile(name, mode);
      document.getElementById("dialog-new-file").classList.add("hidden");
      showEditor(file.id);
    });
  }

  // ===== Editor =====
  function _bindEditorEvents() {
    document.getElementById("btn-back").addEventListener("click", function () {
      _saveCurrentFile();
      showFileList();
    });
    document.getElementById("btn-save").addEventListener("click", function () {
      _saveCurrentFile();
      _showToast("保存しました");
    });
    document.getElementById("btn-clear").addEventListener("click", function () {
      if (confirm("すべてのデータをクリアしますか？")) {
        _currentFile.rows = [createRow()];
        _recalcAndRender();
        _saveCurrentFile();
      }
    });
    document.getElementById("btn-add-row").addEventListener("click", function () {
      _currentFile.rows.push(createRow());
      _recalcAndRender();
      _saveCurrentFile();
    });

    // Mode toggle
    document.getElementById("btn-mode-m").addEventListener("click", function () {
      if (_currentFile) { _currentFile.displayMode = "m"; _updateModeButtons(); _recalcAndRender(); _saveCurrentFile(); }
    });
    document.getElementById("btn-mode-mm").addEventListener("click", function () {
      if (_currentFile) { _currentFile.displayMode = "mm"; _updateModeButtons(); _recalcAndRender(); _saveCurrentFile(); }
    });
  }

  function _updateModeButtons() {
    const isM = _currentFile.displayMode === "m";
    document.getElementById("btn-mode-m").classList.toggle("active", isM);
    document.getElementById("btn-mode-mm").classList.toggle("active", !isM);
  }

  // ===== Table Rendering =====
  function _recalcAndRender() {
    if (!_currentFile) return;
    _currentFile.rows = recalculate(_currentFile.rows);
    _renderTable();
  }

  function _renderTable() {
    const tbody = document.getElementById("table-body");
    const mode = _currentFile.displayMode;
    const rows = _currentFile.rows;

    let html = "";
    for (let i = 0; i < rows.length; i++) {
      const row = rows[i];
      const manual = row.manualFields || [];

      // Insert zone before each row (except first)
      if (i > 0) {
        html += '<tr class="insert-zone-row"><td colspan="7"><div class="insert-zone" data-index="' + i + '"></div></td></tr>';
      }

      html += '<tr data-row-id="' + row.id + '">';

      // Point (text input, not keypad)
      html += '<td class="sticky-col"><input class="cell-input" type="text" data-field="point" data-row="' + i + '" value="' + _escapeHtml(row.point) + '"></td>';

      // BS
      html += _renderCell(row, i, "bs", manual, mode);

      // IH (computed, readonly)
      html += '<td class="computed">' + formatValue(row.ih, mode) + '</td>';

      // FS
      html += _renderCell(row, i, "fs", manual, mode);

      // GH (can be manual or computed)
      html += _renderCell(row, i, "gh", manual, mode);

      // FH
      html += _renderCell(row, i, "fh", manual, mode);

      // Diff (computed, readonly)
      const diffStr = formatValue(row.diff, mode);
      let diffClass = "computed";
      if (row.diff !== null) {
        diffClass += row.diff > 0 ? " diff-plus" : row.diff < 0 ? " diff-minus" : "";
      }
      html += '<td class="' + diffClass + '">' + diffStr + '</td>';

      html += '</tr>';
    }

    tbody.innerHTML = html;
    _bindCellEvents();
  }

  function _renderCell(row, rowIndex, field, manual, mode) {
    const value = row[field];
    const display = formatValue(value, mode);
    const isManual = manual.includes(field);
    const cls = "cell-input" + (isManual ? " manual" : "");
    return '<td><div class="' + cls + '" data-field="' + field + '" data-row="' + rowIndex + '" tabindex="0">' + display + '</div></td>';
  }

  function _bindCellEvents() {
    // Numeric cells → open keypad
    document.querySelectorAll("#table-body .cell-input[data-field]:not([type='text'])").forEach(function (el) {
      el.addEventListener("click", function () {
        const rowIndex = parseInt(this.dataset.row);
        const field = this.dataset.field;
        const row = _currentFile.rows[rowIndex];
        const mode = _currentFile.displayMode;
        const currentDisplay = formatValue(row[field], mode);

        // Scroll target cell into view above keypad
        this.scrollIntoView({ block: "center", behavior: "smooth" });

        Keypad.show(currentDisplay, function (resultStr) {
          const parsed = parseInput(resultStr, mode);
          row[field] = parsed;

          // Update manualFields
          const mf = row.manualFields || [];
          const idx = mf.indexOf(field);
          if (parsed !== null) {
            if (idx === -1) mf.push(field);
          } else {
            if (idx !== -1) mf.splice(idx, 1);
          }
          row.manualFields = mf;

          _recalcAndRender();
          _saveCurrentFile();
        });
      });
    });

    // Point text input
    document.querySelectorAll('#table-body input[data-field="point"]').forEach(function (el) {
      el.addEventListener("change", function () {
        const rowIndex = parseInt(this.dataset.row);
        _currentFile.rows[rowIndex].point = this.value;
        _saveCurrentFile();
      });
    });

    // Insert zones
    document.querySelectorAll(".insert-zone").forEach(function (el) {
      el.addEventListener("click", function () {
        const index = parseInt(this.dataset.index);
        _currentFile.rows.splice(index, 0, createRow());
        _recalcAndRender();
        _saveCurrentFile();
      });
    });

    // Long press or left-swipe to delete
    document.querySelectorAll("#table-body tr[data-row-id]").forEach(function (el) {
      let timer = null;
      let startX = 0;
      let startY = 0;
      el.addEventListener("touchstart", function (e) {
        startX = e.touches[0].clientX;
        startY = e.touches[0].clientY;
        timer = setTimeout(function () {
          const rowId = el.dataset.rowId;
          _deleteRowById(rowId);
        }, 600);
      });
      el.addEventListener("touchmove", function (e) {
        clearTimeout(timer);
        const dx = e.touches[0].clientX - startX;
        const dy = Math.abs(e.touches[0].clientY - startY);
        // Left swipe: horizontal distance > 80px, more horizontal than vertical
        if (dx < -80 && dy < 40) {
          _deleteRowById(el.dataset.rowId);
        }
      });
      el.addEventListener("touchend", function () { clearTimeout(timer); });
    });
  }

  function _deleteRowById(rowId) {
    const index = _currentFile.rows.findIndex(function (r) { return r.id === rowId; });
    if (index === -1) return;
    if (_currentFile.rows.length <= 1) { _showToast("最後の行は削除できません"); return; }

    _deletedRow = Object.assign({}, _currentFile.rows[index]);
    _deletedRow.manualFields = _currentFile.rows[index].manualFields.slice();
    _deletedRowIndex = index;

    _currentFile.rows.splice(index, 1);
    _recalcAndRender();
    _saveCurrentFile();

    _showToast("行を削除しました", function () {
      // Undo
      _currentFile.rows.splice(_deletedRowIndex, 0, _deletedRow);
      _deletedRow = null;
      _deletedRowIndex = -1;
      _recalcAndRender();
      _saveCurrentFile();
    });
  }

  // ===== Auto-save =====
  function _bindAutoSave() {
    document.addEventListener("visibilitychange", function () {
      if (document.visibilityState === "hidden") {
        _saveCurrentFile();
      }
    });
  }

  function _saveCurrentFile() {
    if (_currentFile) {
      saveFile(_currentFile);
    }
  }

  // ===== Toast =====
  function _showToast(message, onUndo) {
    const toast = document.getElementById("toast");
    if (_toastTimer) clearTimeout(_toastTimer);
    let html = message;
    if (onUndo) {
      html += '<button id="toast-undo">元に戻す</button>';
    }
    toast.innerHTML = html;
    toast.classList.remove("hidden");

    if (onUndo) {
      document.getElementById("toast-undo").addEventListener("click", function () {
        onUndo();
        toast.classList.add("hidden");
        clearTimeout(_toastTimer);
      });
    }

    _toastTimer = setTimeout(function () {
      toast.classList.add("hidden");
      _deletedRow = null;
    }, 5000);
  }

  // ===== Util =====
  function _escapeHtml(str) {
    if (!str) return "";
    return str.replace(/&/g, "&amp;").replace(/</g, "&lt;").replace(/>/g, "&gt;").replace(/"/g, "&quot;");
  }

  return { init: init };
})();

document.addEventListener("DOMContentLoaded", App.init);
```

```javascript
// sw.js — Cache-first Service Worker
const CACHE_NAME = "kikikou-fieldbook-v1";
const ASSETS = [
  "./index.html",
  "./style.css",
  "./calc.js",
  "./storage.js",
  "./keypad.js",
  "./app.js",
  "./manifest.json",
  "./icon-192.png",
  "./icon-512.png",
];

self.addEventListener("install", function (event) {
  event.waitUntil(
    caches.open(CACHE_NAME).then(function (cache) {
      return cache.addAll(ASSETS);
    })
  );
  self.skipWaiting();
});

self.addEventListener("activate", function (event) {
  event.waitUntil(
    caches.keys().then(function (keys) {
      return Promise.all(
        keys.filter(function (k) { return k !== CACHE_NAME; })
            .map(function (k) { return caches.delete(k); })
      );
    })
  );
  self.clients.claim();
});

self.addEventListener("fetch", function (event) {
  event.respondWith(
    caches.match(event.request).then(function (cached) {
      return cached || fetch(event.request);
    })
  );
});
```
